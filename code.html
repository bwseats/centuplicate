                        // controller function
                        // gets user input from fields
                        function getValues() {

                            // get values
                            let startValue = document.getElementById('startValue').value;
                            let endValue = document.getElementById('endValue').value;
                        
                            // parse inputs as numbers
                            startValue = parseInt(startValue);
                            endValue = parseInt(endValue);
                        
                            // verify inputs are numbers
                            if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
                            
                                // if so, generate numbers
                                let numbersArray = generateNumbers(startValue, endValue);
                            
                                // display on page
                                displayNumbers(numbersArray);
                            
                            } else {

                                // else, notify user
                            
                                Swal.fire(
                                    {
                                        icon: 'error',
                                        title: 'Oops!',
                                        text: 'Only integers are allowed'
                                    }
                                );
                            }
                        }

                        // logic function
                        // generates numbers in a range
                        function generateNumbers(start, end) {
                        
                            let numbers = [];
                        
                            for (let i = start; i <= end; i++) {
                                numbers.push(i);
                            }
                            return numbers;
                        }

                        // view function
                        // displays results
                        function displayNumbers(numbersArray) {
                            let tableBody = document.getElementById('results');
                        
                            let tableHTML = "";
                        
                            for (let i = 0; i < numbersArray.length; i++) {
                                let value = numbersArray[i];
                                let className = value % 2 == 0 ? 'even' : 'odd';
                            
                                if (i % 5 == 0) {
                                    tableHTML += ''
                                }

                                let tableRow = `${value}`;
                            
                            
                                tableHTML += tableRow
                            
                                if ((i + 1) % 5 == 0) {
                                    tableHTML += ''
                                }
                            }
                        
                            tableBody.innerHTML = tableHTML;
                        }

The code is structured in three functions:

getValues()

This function gets the value from the corresponding HTML elements and stores them in variables startValue and endValue. It then checks to make sure the numbers are indeed integers and, if so, passes them as parameters to the next function:

generateNumbers()

This function simply uses a for loop to iterate through the user-specified range, adding each number in that range to an array. It then returns that array as a variable numbersArray, which is passed as a parameter to the final function:

displayNumbers()

This function checks whether each item in numbersArray is even or odd, and updates className, to be referenced later by the CSS and make all even numbers bold. After this, the numbers are placed into rows of five and injected into the HTML to update the table and display the results.
